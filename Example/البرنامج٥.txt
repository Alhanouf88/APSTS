# ادخل قيمة للمتغير “رقم” لمعرفة اذا كان العدد اولي ام لا

# الاعداد الاولية اكبر من ١
اذا رقم > ١ :
	كرر ن داخل نطاق(٢،رقم):
		اذا (رقم٪ن) == ٠:
			اطبع (“العدد” ، رقم ، “ليس عدد اولي”)
			اطبع (“العدد”، ن، “ضرب العدد” ، رقم//ن، “يساوي” ، رقم)
			توقف
	والا:
		اطبع (“العدد” ، رقم، “عدد اولي”)
